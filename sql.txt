
create table events (
	event_name VARCHAR(200),
	event_start INT,
	event_end INT,
	event_lat decimal(10,2),
	event_long decimal(10,2),
	event_loc varchar(300),
	event_desc TEXT,
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY(ID)
);

create table event_comments (
	event_id INT,
	comment TEXT,
	date_made INT,
	user_id INT,
	active INT,
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY (ID)
);

create table users (
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY(ID)
);

create table talks (
	talk_title TINYTEXT,
	talk_desc TEXT,
	speaker TINYTEXT,
	slides_link TINYTEXT,
	date_given INT,
	event_id INT,
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY(ID)
);

create table talk_comments (
	talk_id INT,
	rating INT,
	comment TEXT,
	date_made INT,
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY (ID)
);

create table user (
	username VARCHAR(100),
	password VARCHAR(32),
	email VARCHAR(200),
	last_login INT,
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY(ID)
);

create table user_admin (
	uid INT,
	rid INT,
	rtype VARCHAR(10),
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY(ID)
);

create table talk_cat (
	talk_id INT,
	cat_id INT,
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY(ID)
);
create table categories (
	cat_title VARCHAR(200),
	cat_desc TEXT,
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY(ID)
);

create table tz (
	offset INT,
	cont VARCHAR(50),
	area VARCHAR(70),
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY(ID)
);

create table user_attend (
	uid INT,
	eid INT,
	ID INT NOT NULL AUTO_INCREMENT,
	PRIMARY KEY(ID)
);
